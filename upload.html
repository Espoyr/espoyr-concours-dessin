import { serve } from "https://deno.land/std@0.168.0/http/server.ts" import { createClient } from "https://esm.sh/@supabase/supabase-js@2" const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", } serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: corsHeaders }) } try { const body = await req.json() const email = String(body.email || "").trim().toLowerCase() if (!email) throw new Error("Email obligatoire") const base = Math.floor(100000 + Math.random() * 900000).toString() const somme = base.split("").reduce((t, c) => t + Number(c), 0) const code = base + (somme % 10) const expiration = "2026-10-15 21:59:59" const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ) const { data: ancienneLigne, error: searchError } = await supabase .from("connexions") .select("id") .eq("email", email) .maybeSingle() if (searchError) throw searchError if (ancienneLigne) { const { error: updateError } = await supabase .from("connexions") .update({ code_connexion: code, valide_jusqua: expiration, }) .eq("id", ancienneLigne.id) if (updateError) throw updateError } else { const { error: insertError } = await supabase .from("connexions") .insert({ email: email, code_connexion: code, valide_jusqua: expiration, }) if (insertError) throw insertError } const mail = await fetch("https://api.brevo.com/v3/smtp/email", { method: "POST", headers: { accept: "application/json", "api-key": Deno.env.get("BREVO_API_KEY")!, "content-type": "application/json", }, body: JSON.stringify({ sender: { name: "Concours Dessin Espoyr", email: "espoyr.asbl@gmail.com", }, to: [{ email: email }], subject: "Votre code de connexion - Concours Espoyr", htmlContent: `

Concours de dessin Espoyr

Votre code de connexion :

${code}

Ce code est valable jusqu'au ${expiration}.

`, }), }) if (!mail.ok) throw new Error(await mail.text()) return new Response(JSON.stringify({ success: true, message: "Code envoyé", email: email }), { status: 200, headers: { ...corsHeaders, "Content-Type": "application/json" }, }) } catch (e) { return new Response(JSON.stringify({ success: false, error: String(e) }), { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" }, }) } })
